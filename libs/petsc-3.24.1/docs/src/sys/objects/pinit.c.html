 85: /*@C
 86:   PetscInitializeNoPointers - Calls PetscInitialize() from C/C++ without the pointers to argc and args

 88:   Collective, No Fortran Support

 90:   Input Parameters:
 91: + argc     - number of args
 92: . args     - array of command line arguments
 93: . filename - optional name of the program file, pass `NULL` to ignore
 94: - help     - optional help, pass `NULL` to ignore

 96:   Level: advanced

 98:   Notes:
 99:   this is called only by the PETSc Julia interface. Even though it might start MPI it sets the flag to
100:   indicate that it did NOT start MPI so that the `PetscFinalize()` does not end MPI, thus allowing `PetscInitialize()` to
101:   be called multiple times from Julia without the problem of trying to initialize MPI more than once.

103:   Developer Notes:
104:   Turns off PETSc signal handling to allow Julia to manage signals

106: .seealso: `PetscInitialize()`, `PetscInitializeFortran()`, `PetscInitializeNoArguments()`
107: */
108: PetscErrorCode PetscInitializeNoPointers(int argc, char **args, const char *filename, const char *help)
109: {
110:   int    myargc = argc;
111:   char **myargs = args;

113:   PetscFunctionBegin;
114:   PetscCall(PetscInitialize(&myargc, &myargs, filename, help));
115:   PetscCall(PetscPopSignalHandler());
116:   PetscBeganMPI = PETSC_FALSE;
117:   PetscFunctionReturn(PETSC_SUCCESS);
118: }

185: PETSC_INTERN void MPIAPI MPIU_MaxSum_Local(void *in, void *out, PetscMPIInt *cnt, MPI_Datatype *datatype)
186: {
187:   PetscFunctionBegin;
188:   if (*datatype == MPIU_INT_MPIINT && PetscDefined(USE_64BIT_INDICES)) {
189: #if defined(PETSC_USE_64BIT_INDICES)
190:     struct petsc_mpiu_int_mpiint *xin = (struct petsc_mpiu_int_mpiint *)in, *xout = (struct petsc_mpiu_int_mpiint *)out;
191:     PetscMPIInt                   count = *cnt;

193:     for (PetscMPIInt i = 0; i < count; i++) {
194:       xout[i].a = PetscMax(xout[i].a, xin[i].a);
195:       xout[i].b += xin[i].b;
196:     }
197: #endif
198:   } else if (*datatype == MPIU_2INT || *datatype == MPIU_INT_MPIINT) {
199:     PetscInt   *xin = (PetscInt *)in, *xout = (PetscInt *)out;
200:     PetscMPIInt count = *cnt;

202:     for (PetscMPIInt i = 0; i < count; i++) {
203:       xout[2 * i] = PetscMax(xout[2 * i], xin[2 * i]);
204:       xout[2 * i + 1] += xin[2 * i + 1];
205:     }
206:   } else {
207:     PetscErrorCode ierr = (*PetscErrorPrintf)("Can only handle MPIU_2INT and MPIU_INT_MPIINT data types");
208:     (void)ierr;
209:     PETSCABORT(MPI_COMM_SELF, PETSC_ERR_ARG_WRONG);
210:   }
211:   PetscFunctionReturnVoid();
212: }

214: /*@
215:   PetscMaxSum - Returns the max of the first entry over all MPI processes and the sum of the second entry.

217:   Collective

219:   Input Parameters:
220: + comm  - the communicator
221: - array - an arry of length 2 times `size`, the number of MPI processes

223:   Output Parameters:
224: + max - the maximum of `array[2*rank]` over all MPI processes
225: - sum - the sum of the `array[2*rank + 1]` over all MPI processes

227:   Level: developer

229: .seealso: `PetscInitialize()`
230: @*/
231: PetscErrorCode PetscMaxSum(MPI_Comm comm, const PetscInt array[], PetscInt *max, PetscInt *sum)
232: {
233:   PetscFunctionBegin;
234: #if defined(PETSC_HAVE_MPI_REDUCE_SCATTER_BLOCK)
235:   {
236:     struct {
237:       PetscInt max, sum;
238:     } work;
239:     PetscCallMPI(MPI_Reduce_scatter_block((void *)array, &work, 1, MPIU_2INT, MPIU_MAXSUM_OP, comm));
240:     *max = work.max;
241:     *sum = work.sum;
242:   }
243: #else
244:   {
245:     PetscMPIInt size, rank;
246:     struct {
247:       PetscInt max, sum;
248:     } *work;
249:     PetscCallMPI(MPI_Comm_size(comm, &size));
250:     PetscCallMPI(MPI_Comm_rank(comm, &rank));
251:     PetscCall(PetscMalloc1(size, &work));
252:     PetscCallMPI(MPIU_Allreduce((void *)array, work, size, MPIU_2INT, MPIU_MAXSUM_OP, comm));
253:     *max = work[rank].max;
254:     *sum = work[rank].sum;
255:     PetscCall(PetscFree(work));
256:   }
257: #endif
258:   PetscFunctionReturn(PETSC_SUCCESS);
259: }

271: {
272:   PetscMPIInt i, count = *cnt;

274:   PetscFunctionBegin;
275:   if (*datatype == MPIU_REAL) {
276:     PetscReal *xin = (PetscReal *)in, *xout = (PetscReal *)out;
277:     for (i = 0; i < count; i++) xout[i] += xin[i];
278:   }
279:   #if defined(PETSC_HAVE_COMPLEX)
280:   else if (*datatype == MPIU_COMPLEX) {
281:     PetscComplex *xin = (PetscComplex *)in, *xout = (PetscComplex *)out;
282:     for (i = 0; i < count; i++) xout[i] += xin[i];
283:   }
284:   #endif
285:   #if defined(PETSC_HAVE_REAL___FLOAT128) && !defined(PETSC_SKIP_REAL___FLOAT128)
286:   else if (*datatype == MPIU___FLOAT128) {
287:     __float128 *xin = (__float128 *)in, *xout = (__float128 *)out;
288:     for (i = 0; i < count; i++) xout[i] += xin[i];
289:     #if defined(PETSC_HAVE_COMPLEX)
290:   } else if (*datatype == MPIU___COMPLEX128) {
291:     __complex128 *xin = (__complex128 *)in, *xout = (__complex128 *)out;
292:     for (i = 0; i < count; i++) xout[i] += xin[i];
293:     #endif
294:   }
295:   #endif
296:   #if defined(PETSC_HAVE_REAL___FP16) && !defined(PETSC_SKIP_REAL___FP16)
297:   else if (*datatype == MPIU___FP16) {
298:     __fp16 *xin = (__fp16 *)in, *xout = (__fp16 *)out;
299:     for (i = 0; i < count; i++) xout[i] = (__fp16)(xin[i] + xout[i]);
300:   }
301:   #endif
302:   else {
303:   #if (!defined(PETSC_HAVE_REAL___FLOAT128) || defined(PETSC_SKIP_REAL___FLOAT128)) && (!defined(PETSC_HAVE_REAL___FP16) || defined(PETSC_SKIP_REAL___FP16))
304:     PetscCallAbort(MPI_COMM_SELF, (*PetscErrorPrintf)("Can only handle MPIU_REAL or MPIU_COMPLEX data types"));
305:   #elif !defined(PETSC_HAVE_REAL___FP16) || defined(PETSC_SKIP_REAL___FP16)
306:     PetscCallAbort(MPI_COMM_SELF, (*PetscErrorPrintf)("Can only handle MPIU_REAL, MPIU_COMPLEX, MPIU___FLOAT128, or MPIU___COMPLEX128 data types"));
307:   #elif !defined(PETSC_HAVE_REAL___FLOAT128) || defined(PETSC_SKIP_REAL___FLOAT128)
308:     PetscCallAbort(MPI_COMM_SELF, (*PetscErrorPrintf)("Can only handle MPIU_REAL, MPIU_COMPLEX, or MPIU___FP16 data types"));
309:   #else
310:     PetscCallAbort(MPI_COMM_SELF, (*PetscErrorPrintf)("Can only handle MPIU_REAL, MPIU_COMPLEX, MPIU___FLOAT128, MPIU___COMPLEX128, or MPIU___FP16 data types"));
311:   #endif
312:     PETSCABORT(MPI_COMM_SELF, PETSC_ERR_ARG_WRONG);
313:   }
314:   PetscFunctionReturnVoid();
315: }

367: /*
368:    Private routine to delete internal tag/name counter storage when a communicator is freed.

370:    This is called by MPI, not by users. This is called by MPI_Comm_free() when the communicator that has this  data as an attribute is freed.

372:    Note: this is declared extern "C" because it is passed to MPI_Comm_create_keyval()

374: */
375: PETSC_EXTERN PetscMPIInt MPIAPI Petsc_Counter_Attr_DeleteFn(MPI_Comm comm, PetscMPIInt keyval, void *count_val, void *extra_state)
376: {
377:   PetscCommCounter      *counter = (PetscCommCounter *)count_val;
378:   struct PetscCommStash *comms   = counter->comms, *pcomm;

380:   PetscFunctionBegin;
381:   PetscCallReturnMPI(PetscInfo(NULL, "Deleting counter data in an MPI_Comm %ld\n", (long)comm));
382:   PetscCallReturnMPI(PetscFree(counter->iflags));
383:   while (comms) {
384:     PetscCallMPIReturnMPI(MPI_Comm_free(&comms->comm));
385:     pcomm = comms;
386:     comms = comms->next;
387:     PetscCallReturnMPI(PetscFree(pcomm));
388:   }
389:   PetscCallReturnMPI(PetscFree(counter));
390:   PetscFunctionReturn(MPI_SUCCESS);
391: }

555:   after `PetscInitialize()` is called but before `PetscFinalize()`.

557:   Not Collective, No Fortran Support

559:   Output Parameter:

562:   Level: intermediate

585:   for (i = 0; i < argc - 1; i++) PetscCall(PetscStrallocpy(PetscGlobalArgs[i + 1], &(*args)[i]));

1416: /*@
1417:   PetscFinalize - Checks for options to be called at the conclusion of a PETSc program and frees any remaining PETSc objects and data structures.
1418:   of the program. Automatically calls `MPI_Finalize()` if the user had not called `MPI_Init()` before calling `PetscInitialize()`.

1420:   Collective on `PETSC_COMM_WORLD`

1422:   Options Database Keys:
1423: + -options_view                    - Calls `PetscOptionsView()`
1424: . -options_left                    - Prints unused options that remain in the database
1425: . -objects_dump [all]              - Prints list of objects allocated by the user that have not been freed, the option all cause all outstanding objects to be listed
1426: . -mpidump                         - Calls PetscMPIDump()
1427: . -malloc_dump <optional filename> - Calls `PetscMallocDump()`, displays all memory allocated that has not been freed
1428: . -memory_view                     - Prints total memory usage
1429: - -malloc_view <optional filename> - Prints list of all memory allocated and in what functions

1431:   Level: beginner

1433:   Note:
1434:   See `PetscInitialize()` for other runtime options.

1436:   You can call `PetscInitialize()` after `PetscFinalize()` but only with MPI-Uni or if you called `MPI_Init()` before ever calling `PetscInitialize()`.

1438: .seealso: `PetscInitialize()`, `PetscOptionsView()`, `PetscMallocDump()`, `PetscMPIDump()`, `PetscEnd()`
1439: @*/

1442:   PetscMPIInt rank;
1443:   PetscInt    nopt;
1444:   PetscBool   flg1 = PETSC_FALSE, flg2 = PETSC_FALSE, flg3 = PETSC_FALSE;
1445:   PetscBool   flg;
1446:   char        mname[PETSC_MAX_PATH_LEN];

1448:   PetscFunctionBegin;
1449:   PetscCheck(PetscInitializeCalled, PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "PetscInitialize() must be called before PetscFinalize()");
1450:   PetscCall(PetscInfo(NULL, "PetscFinalize() called\n"));

1452:   PetscCall(PetscOptionsHasName(NULL, NULL, "-mpi_linear_solver_server", &flg));
1453:   if (PetscDefined(USE_SINGLE_LIBRARY) && flg) PetscCall(PCMPIServerEnd());

1455:   PetscCall(PetscFreeAlign(PetscGlobalArgsFortran, 0, NULL, NULL));
1456:   PetscGlobalArgc = 0;
1457:   PetscGlobalArgs = NULL;

1459:   /* Clean up Garbage automatically on COMM_SELF and COMM_WORLD at finalize */
1460:   {
1461:     union
1462:     {
1463:       MPI_Comm comm;
1464:       void    *ptr;
1465:     } ucomm;
1466:     PetscMPIInt flg;

1469:     PetscCallMPI(MPI_Comm_get_attr(PETSC_COMM_SELF, Petsc_InnerComm_keyval, &ucomm, &flg));
1470:     if (flg) PetscCallMPI(MPI_Comm_get_attr(ucomm.comm, Petsc_Garbage_HMap_keyval, &tmp, &flg));
1471:     if (flg) PetscCall(PetscGarbageCleanup(PETSC_COMM_SELF));
1472:     PetscCallMPI(MPI_Comm_get_attr(PETSC_COMM_WORLD, Petsc_InnerComm_keyval, &ucomm, &flg));
1473:     if (flg) PetscCallMPI(MPI_Comm_get_attr(ucomm.comm, Petsc_Garbage_HMap_keyval, &tmp, &flg));
1474:     if (flg) PetscCall(PetscGarbageCleanup(PETSC_COMM_WORLD));
1475:   }

1477:   PetscCallMPI(MPI_Comm_rank(PETSC_COMM_WORLD, &rank));
1478: #if defined(PETSC_HAVE_ADIOS)
1479:   PetscCallExternal(adios_read_finalize_method, ADIOS_READ_METHOD_BP_AGGREGATE);
1480:   PetscCallExternal(adios_finalize, rank);
1481: #endif
1482:   PetscCall(PetscOptionsHasName(NULL, NULL, "-citations", &flg));
1483:   if (flg) {
1484:     char *cits, filename[PETSC_MAX_PATH_LEN];
1485:     FILE *fd = PETSC_STDOUT;

1487:     PetscCall(PetscOptionsGetString(NULL, NULL, "-citations", filename, sizeof(filename), NULL));
1488:     if (filename[0]) PetscCall(PetscFOpen(PETSC_COMM_WORLD, filename, "w", &fd));
1489:     PetscCall(PetscSegBufferGet(PetscCitationsList, 1, &cits));

1493:     PetscCall(PetscFPrintf(PETSC_COMM_WORLD, fd, "===========================================================================\n"));
1494:     PetscCall(PetscFPrintf(PETSC_COMM_WORLD, fd, "%s", cits));
1495:     PetscCall(PetscFPrintf(PETSC_COMM_WORLD, fd, "===========================================================================\n"));
1496:     PetscCall(PetscFClose(PETSC_COMM_WORLD, fd));
1497:     PetscCall(PetscFree(cits));
1498:   }
1499:   PetscCall(PetscSegBufferDestroy(&PetscCitationsList));

1501: #if defined(PETSC_SERIALIZE_FUNCTIONS)
1502:   PetscCall(PetscFPTDestroy());
1503: #endif

1505: #if defined(PETSC_HAVE_X)
1506:   flg1 = PETSC_FALSE;
1507:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-x_virtual", &flg1, NULL));
1508:   if (flg1) {
1509:     /*  this is a crude hack, but better than nothing */

1512: #endif

1514: #if !defined(PETSC_HAVE_THREADSAFETY)
1515:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-memory_view", &flg2, NULL));

1517: #endif

1527:   }

1529:   if (PetscDefined(USE_LOG) && PetscDefined(HAVE_MPE)) {
1530:     mname[0] = 0;
1531:     PetscCall(PetscOptionsGetString(NULL, NULL, "-log_mpe", mname, sizeof(mname), &flg1));
1532:     if (flg1) PetscCall(PetscLogMPEDump(mname[0] ? mname : NULL));
1533:   }

1535: #if defined(PETSC_HAVE_KOKKOS)
1536:   // Free PETSc/kokkos stuff before the potentially non-null PETSc default gpu stream is destroyed by PetscObjectRegisterDestroyAll
1537:   if (PetscKokkosInitialized) {
1538:     PetscCall(PetscKokkosFinalize_Private());

1541: #endif

1551:     PetscLogGpuTimeFlag = PETSC_FALSE;

1553:     // Free any objects created by the last block of code.
1554:     PetscCall(PetscObjectRegisterDestroyAll());

1556:     mname[0] = 0;
1557:     PetscCall(PetscOptionsGetString(NULL, NULL, "-log_all", mname, sizeof(mname), &flg1));
1558:     PetscCall(PetscOptionsGetString(NULL, NULL, "-log", mname, sizeof(mname), &flg2));
1559:     if (flg1 || flg2) PetscCall(PetscLogDump(mname));
1560:   }

1562:   flg1 = PETSC_FALSE;
1563:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-no_signal_handler", &flg1, NULL));
1564:   if (!flg1) PetscCall(PetscPopSignalHandler());
1565:   flg1 = PETSC_FALSE;
1566:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-mpidump", &flg1, NULL));
1567:   if (flg1) PetscCall(PetscMPIDump(stdout));
1568:   flg1 = PETSC_FALSE;
1569:   flg2 = PETSC_FALSE;
1570:   /* preemptive call to avoid listing this option in options table as unused */
1571:   PetscCall(PetscOptionsHasName(NULL, NULL, "-malloc_dump", &flg1));
1572:   PetscCall(PetscOptionsHasName(NULL, NULL, "-objects_dump", &flg1));
1573:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-options_view", &flg2, NULL));

1575:   if (flg2) PetscCall(PetscOptionsView(NULL, PETSC_VIEWER_STDOUT_WORLD));

1577:   /* to prevent PETSc -options_left from warning */
1578:   PetscCall(PetscOptionsHasName(NULL, NULL, "-nox", &flg1));
1579:   PetscCall(PetscOptionsHasName(NULL, NULL, "-nox_warning", &flg1));

1581:   flg3 = PETSC_FALSE; /* default value is required */
1582:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-options_left", &flg3, &flg1));
1583:   if (!flg1) flg3 = PETSC_TRUE;
1584:   if (flg3) {
1585:     if (!flg2 && flg1) { /* have not yet printed the options */
1586:       PetscCall(PetscOptionsView(NULL, PETSC_VIEWER_STDOUT_WORLD));
1587:     }
1588:     PetscCall(PetscOptionsAllUsed(NULL, &nopt));
1589:     if (nopt) {
1590:       PetscCall(PetscPrintf(PETSC_COMM_WORLD, "WARNING! There are options you set that were not used!\n"));

1593:         PetscCall(PetscPrintf(PETSC_COMM_WORLD, "There is one unused database option. It is:\n"));
1594:       } else {
1595:         PetscCall(PetscPrintf(PETSC_COMM_WORLD, "There are %" PetscInt_FMT " unused database options. They are:\n", nopt));
1596:       }
1597:     } else if (flg3 && flg1) {
1598:       PetscCall(PetscPrintf(PETSC_COMM_WORLD, "There are no unused options.\n"));
1599:     }
1600:     PetscCall(PetscOptionsLeft(NULL));
1601:   }

1603: #if defined(PETSC_HAVE_SAWS)
1604:   if (!PetscGlobalRank) {
1605:     PetscCall(PetscStackSAWsViewOff());

1608: #endif

1610:   /*
1611:        List all objects the user may have forgot to free
1612:   */
1613:   if (PetscDefined(USE_LOG) && PetscObjectsLog) {

1615:     if (flg1) {
1616:       MPI_Comm local_comm;
1617:       char     string[64];

1619:       PetscCall(PetscOptionsGetString(NULL, NULL, "-objects_dump", string, sizeof(string), NULL));
1620:       PetscCallMPI(MPI_Comm_dup(PETSC_COMM_WORLD, &local_comm));
1621:       PetscCall(PetscSequentialPhaseBegin_Private(local_comm, 1));
1622:       PetscCall(PetscObjectsDump(stdout, (string[0] == 'a') ? PETSC_TRUE : PETSC_FALSE));
1623:       PetscCall(PetscSequentialPhaseEnd_Private(local_comm, 1));
1624:       PetscCallMPI(MPI_Comm_free(&local_comm));
1625:     }
1626:   }

1628:   PetscObjectsCounts    = 0;
1629:   PetscObjectsMaxCounts = 0;
1630:   PetscCall(PetscFree(PetscObjects));

1632:   /*
1633:      Destroy any packages that registered a finalize
1634:   */
1635:   PetscCall(PetscRegisterFinalizeAll());

1637:   PetscCall(PetscLogFinalize());

1639:   /*
1640:      Print PetscFunctionLists that have not been properly freed
1641:   */
1642:   if (PetscPrintFunctionList) PetscCall(PetscFunctionListPrintAll());

1644:   if (petsc_history) {
1645:     PetscCall(PetscCloseHistoryFile(&petsc_history));
1646:     petsc_history = NULL;
1647:   }
1648:   PetscCall(PetscOptionsHelpPrintedDestroy(&PetscOptionsHelpPrintedSingleton));
1649:   PetscCall(PetscInfoDestroy());

1651: #if !defined(PETSC_HAVE_THREADSAFETY)
1652:   if (!(PETSC_RUNNING_ON_VALGRIND)) {
1653:     char  fname[PETSC_MAX_PATH_LEN];
1654:     char  sname[PETSC_MAX_PATH_LEN];
1655:     FILE *fd;
1656:     int   err;

1658:     flg2 = PETSC_FALSE;
1659:     flg3 = PETSC_FALSE;
1660:     if (PetscDefined(USE_DEBUG)) PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_test", &flg2, NULL));

1666:       fd = fopen(sname, "w");
1667:       PetscCheck(fd, PETSC_COMM_SELF, PETSC_ERR_FILE_OPEN, "Cannot open log file: %s", sname);
1668:       PetscCall(PetscMallocDump(fd));
1669:       err = fclose(fd);
1670:       PetscCheck(!err, PETSC_COMM_SELF, PETSC_ERR_SYS, "fclose() failed on file");
1671:     } else if (flg1 || flg2 || flg3) {
1672:       MPI_Comm local_comm;

1674:       PetscCallMPI(MPI_Comm_dup(PETSC_COMM_WORLD, &local_comm));
1675:       PetscCall(PetscSequentialPhaseBegin_Private(local_comm, 1));

1684:       fd = fopen(sname, "w");
1685:       PetscCheck(fd, PETSC_COMM_SELF, PETSC_ERR_FILE_OPEN, "Cannot open log file: %s", sname);
1686:       PetscCall(PetscMallocView(fd));
1687:       err = fclose(fd);
1688:       PetscCheck(!err, PETSC_COMM_SELF, PETSC_ERR_SYS, "fclose() failed on file");
1689:     } else if (flg1) {
1690:       MPI_Comm local_comm;

1692:       PetscCallMPI(MPI_Comm_dup(PETSC_COMM_WORLD, &local_comm));
1693:       PetscCall(PetscSequentialPhaseBegin_Private(local_comm, 1));
1694:       PetscCall(PetscMallocView(stdout));

1775:     PetscCheck(!flag, PETSC_COMM_SELF, PETSC_ERR_LIB, "MPI_Finalize() has already been called, even though MPI_Init() was called by PetscInitialize()");
1776:     /* wait until the very last moment to disable error handling */
1777:     PetscErrorHandlingInitialized = PETSC_FALSE;
1778:     PetscCallMPI(MPI_Finalize());
1779:   } else PetscErrorHandlingInitialized = PETSC_FALSE;

1790: */
1791:   PetscCall(PetscMallocClear());
1792:   PetscCall(PetscStackReset());

1794:   PetscInitializeCalled = PETSC_FALSE;
1795:   PetscFinalizeCalled   = PETSC_TRUE;
1796: #if defined(PETSC_USE_COVERAGE)

1802: #endif
1803:   /* To match PetscFunctionBegin() at the beginning of this function */
1804:   PetscStackClearTop;
1805:   return PETSC_SUCCESS;
1806: }

1823:   if (*a - 32 == *b) return 1;
1824:   return 0;
1825: }
1826: #endif

1842: /*
1843:      When count is 1 and dtype == MPIU_INT performs the reduction in PetscInt64 to check for integer overflow
1844: */
1845: PetscMPIInt MPIU_Allreduce_Private(const void *inbuf, void *outbuf, MPIU_Count count, MPI_Datatype dtype, MPI_Op op, MPI_Comm comm)
1846: {
1847:   PetscMPIInt err;
1848:   if (!PetscDefined(USE_64BIT_INDICES) && count == 1 && dtype == MPIU_INT && (op == MPI_SUM || op == MPI_PROD)) {
1849:     PetscInt64 incnt, outcnt;
1850:     void      *inbufd, *outbufd;

1852:     if (inbuf != MPI_IN_PLACE) {
1853:       incnt   = *(PetscInt32 *)inbuf;
1854:       inbufd  = &incnt;
1855:       outbufd = &outcnt;
1856:       err     = MPIU_Allreduce_Count(inbufd, outbufd, count, MPIU_INT64, op, comm);
1857:     } else {
1858:       outcnt  = *(PetscInt32 *)outbuf;
1859:       outbufd = &outcnt;
1860:       err     = MPIU_Allreduce_Count(MPI_IN_PLACE, outbufd, count, MPIU_INT64, op, comm);
1861:     }
1862:     if (!err && outcnt > PETSC_INT_MAX) err = MPI_ERR_OTHER;
1863:     *(PetscInt32 *)outbuf = (PetscInt32)outcnt;
1864:   } else {
1865:     err = MPIU_Allreduce_Count(inbuf, outbuf, count, dtype, op, comm);
1866:   }
1867:   return err;
1868: }

1870: // Check if MPIU_Allreduce() is called on the same filename:lineno and with the same data count across all processes. Error out if otherwise.
1871: PetscErrorCode PetscCheckAllreduceSameLineAndCount_Private(MPI_Comm comm, const char *filename, PetscMPIInt lineno, PetscMPIInt count)
1872: {
1873:   PetscMPIInt rbuf[4];

1875:   PetscFunctionBegin;
1876:   rbuf[0] = lineno;
1877:   rbuf[1] = -rbuf[0];
1878:   rbuf[2] = count;
1879:   rbuf[3] = -rbuf[2];
1880:   PetscCallMPI(MPI_Allreduce(MPI_IN_PLACE, rbuf, 4, MPI_INT, MPI_MAX, comm));

1882:   if (rbuf[0] != -rbuf[1]) {
1883:     size_t      len;
1884:     PetscMPIInt size, rank, ilen, *recvcounts = NULL, *displs = NULL;
1885:     char       *str = NULL, *str0 = NULL;

1887:     PetscCallMPI(MPI_Comm_size(comm, &size));
1888:     PetscCallMPI(MPI_Comm_rank(comm, &rank));
1889:     PetscCall(PetscStrlen(filename, &len));
1890:     len += 128; /* add enough space for the leading and trailing chars in PetscSNPrintf around __FILE__ */
1891:     PetscCall(PetscMalloc1(len, &str));
1892:     PetscCall(PetscSNPrintf(str, len, "                On process %d, %s:%d\n", rank, filename, lineno));
1893:     PetscCall(PetscStrlen(str, &len)); /* string length exclusive of the NULL terminator */
1894:     ilen = (PetscMPIInt)len;
1895:     if (rank == 0) PetscCall(PetscMalloc2(size, &recvcounts, size + 1, &displs));
1896:     PetscCallMPI(MPI_Gather(&ilen, 1, MPI_INT, recvcounts, 1, MPI_INT, 0, comm));
1897:     if (rank == 0) {
1898:       displs[0] = 0;
1899:       for (PetscMPIInt i = 0; i < size; i++) displs[i + 1] = displs[i] + recvcounts[i];
1900:       PetscCall(PetscMalloc1(displs[size], &str0));
1901:     }
1902:     PetscCallMPI(MPI_Gatherv(str, ilen, MPI_CHAR, str0, recvcounts, displs, MPI_CHAR, 0, comm));
1903:     if (rank == 0) str0[displs[size] - 1] = 0; /* replace the ending \n with NULL */
1904:     PetscCall(PetscFree(str));
1905:     if (rank == 0) PetscCall(PetscFree2(recvcounts, displs));
1906:     SETERRQ(comm, PETSC_ERR_PLIB, "MPIU_Allreduce() called in different locations on different processes:\n%s", str0);
1907:   }
1908:   PetscCheck(rbuf[2] == -rbuf[3], comm, PETSC_ERR_PLIB, "MPIU_Allreduce() called with different counts %d on different processes", count);
1909:   PetscFunctionReturn(PETSC_SUCCESS);
1910: }
