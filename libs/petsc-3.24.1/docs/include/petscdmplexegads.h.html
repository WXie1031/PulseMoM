 19: #define PetscCallEGADS(func, args) \
 20:   do { \
 21:     int _status; \
 22:     PetscStackPushExternal(#func); \
 23:     _status = func args; \
 24:     PetscStackPop; \
 25:     PetscCheck(_status >= 0, PETSC_COMM_SELF, PETSC_ERR_LIB, "Error in EGADS call %s() Status %d", #func, (int)_status); \
 26:   } while (0)
