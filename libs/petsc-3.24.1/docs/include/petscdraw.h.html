 72: /*MC
 73:    PetscDrawRealToColor - Maps a real value within an interval to a color.
 74:    The color is an integer value in the range [`PETSC_DRAW_BASIC_COLORS` to 255]
 75:    that can be passed to various drawing routines.

 77:    Synopsis:
 78: #include <petscdraw.h>
 79:    int PetscDrawRealToColor(PetscReal value,PetscReal min,PetscReal max)

 81:    Not Collective

 83:    Input Parameters:
 84: +  value - value to map within the interval [`min`, `max`]
 85: .  min   - lower end of interval
 86: -  max   - upper end of interval

 88:    Returns:
 89:    The result as integer

 91:    Level: intermediate

 93:    Note:
 94:    Values outside the interval [`min`, `max`] are clipped.

 96: .seealso: `PetscDraw`, `PetscDrawPointPixel()`, `PetscDrawPoint()`, `PetscDrawLine()`, `PetscDrawTriangle()`, `PetscDrawRectangle()`
 97: M*/
 98: static inline int PetscDrawRealToColor(PetscReal value, PetscReal min, PetscReal max)
 99: {
100:   value = PetscClipInterval(value, min, max);
101:   return PETSC_DRAW_BASIC_COLORS + (int)((255 - PETSC_DRAW_BASIC_COLORS) * (value - min) / (max - min));
102: }

352: #if PetscDefined(HAVE_X) && PetscDefined(HAVE_SETJMP_H)
353:   #include <setjmp.h>

355: PETSC_EXTERN jmp_buf                                 PetscXIOErrorHandlerJumpBuf;
356: PETSC_EXTERN void                                    PetscXIOErrorHandlerJump(void *);
357: PETSC_EXTERN_TYPEDEF typedef void                    PetscXIOErrorHandlerFn(void *);
358: PETSC_EXTERN_TYPEDEF typedef PetscXIOErrorHandlerFn *PetscXIOErrorHandler;
359: PETSC_EXTERN PetscXIOErrorHandlerFn                 *PetscSetXIOErrorHandler(PetscXIOErrorHandlerFn *);

361:   /*MC
362:    PetscDrawCollectiveBegin - Begins a set of draw operations

364:    Collective

366:    Synopsis:
367: #include <petscdraw.h>
368:     PetscErrorCode PetscDrawCollectiveBegin(PetscDraw draw)

370:    Collective

372:    Input Parameter:
373: .   draw - the draw object

375:    Level: advanced

377:    Notes:
378:    This is a macro that handles its own error checking, it does not return an error code.

380:    The set of operations needs to be ended by a call to `PetscDrawCollectiveEnd()`.

382:    X windows draw operations that are enclosed by these routines handle correctly resizing or closing of
383:    the window without crashing the program.

385:    Developer Note:
386:    This only applies to X windows and so should have a more specific name such as `PetscDrawXCollectiveBegin()`

388: .seealso: `PetscDraw`, `PetscDrawCollectiveEnd()`
389: M*/
390:   #define PetscDrawCollectiveBegin(draw) \
391:     do { \
392:       jmp_buf _Petsc_jmpbuf; \
393:       PetscXIOErrorHandlerFn *volatile _Petsc_xioerrhdl = PETSC_NULLPTR; \
394:       PetscBool _Petsc_isdrawx, _Petsc_xioerr, _Petsc_xioerr_local = PETSC_FALSE; \
395:       PetscCall(PetscObjectTypeCompare((PetscObject)(draw), PETSC_DRAW_X, &_Petsc_isdrawx)); \
396:       if (_Petsc_isdrawx) { \
397:         PetscCall(PetscMemcpy(&_Petsc_jmpbuf, &PetscXIOErrorHandlerJumpBuf, sizeof(_Petsc_jmpbuf))); \
398:         _Petsc_xioerrhdl = PetscSetXIOErrorHandler(PetscXIOErrorHandlerJump); \
399:         if (setjmp(PetscXIOErrorHandlerJumpBuf)) { \
400:           _Petsc_xioerr_local = PETSC_TRUE; \
401:           do { \
402:             PetscDrawCollectiveEnd(draw); \
403:           } \
404:         } \
405:         do { \
406:       } while (0)

408:   /*MC
409:    PetscDrawCollectiveEnd - Ends a set of draw operations begun with `PetscDrawCollectiveBegin()`

411:    Collective

413:    Synopsis:
414: #include <petscdraw.h>
415:     PetscErrorCode PetscDrawCollectiveEnd(PetscDraw draw)

417:    Collective

419:    Input Parameter:
420: .   draw - the draw object

422:    Level: advanced

424:    Notes:
425:    This is a macro that handles its own error checking, it does not return an error code.

427:    X windows draw operations that are enclosed by these routines handle correctly resizing or closing of
428:    the window without crashing the program.

430:    Developer Note:
431:    This only applies to X windows and so should have a more specific name such as `PetscDrawXCollectiveEnd()`

433: .seealso: `PetscDraw`, `PetscDrawCollectiveBegin()`
434: M*/
435:   #define PetscDrawCollectiveEnd(draw) \
